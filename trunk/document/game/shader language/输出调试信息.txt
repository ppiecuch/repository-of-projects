void printShaderInfoLog(GLuint obj)
{
     int infologLength = 0;
     int charsWritten   = 0;
     char *infoLog;

glGetShaderiv(obj, GL_INFO_LOG_LENGTH,&infologLength);

     if (infologLength > 0)
     {
         infoLog = (char *)malloc(infologLength);
         glGetShaderInfoLog(obj, infologLength, &charsWritten, infoLog);
printf("%s\n",infoLog);
         free(infoLog);
     }
}
