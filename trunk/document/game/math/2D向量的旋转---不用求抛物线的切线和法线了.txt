/**
     * 旋转向量
     * 逆时针为正方向
     * @param sin
     * @param cos
     */
    public void Rotate(float sin, float cos){
        float _x = x*cos - y*sin;
        float _y = x*sin + y*cos;
        this.x = _x;
        this.y = _y;
    }
